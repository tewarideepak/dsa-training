ğŸ§ª Example: Abstract Class

#include <iostream>
using namespace std;

class Shape {
public:
    // Pure virtual function (makes the class abstract)
    virtual void draw() = 0;  
};

class Circle : public Shape {
public:
    void draw() override {
        cout << "Drawing Circle" << endl;
    }
};

int main() {
    Shape* s = new Circle();
    s->draw();  // Output: Drawing Circle
    delete s;
}
